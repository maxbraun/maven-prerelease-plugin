16  package net.oneandone.maven.plugins.prerelease;
17  
18  import net.oneandone.maven.plugins.prerelease.core.Archive;
19  import net.oneandone.maven.plugins.prerelease.core.Descriptor;
20  import net.oneandone.maven.plugins.prerelease.core.Prerelease;
21  import net.oneandone.maven.plugins.prerelease.core.WorkingCopy;
22  import net.oneandone.maven.plugins.prerelease.util.Maven;
23  import org.apache.maven.plugins.annotations.Mojo;
24  
25  /**
26   * Checks if there is a prerelease for the last change in your svn working directory, creates one if not.
27   */
28  @Mojo(name = "update")
29  public class Update extends ProjectBase {
30      @Override
31      public void doExecute(Archive archive) throws Exception {
32          WorkingCopy workingCopy;
33          Descriptor descriptor;
34          long revision;
35          Prerelease prerelease;
36          Maven maven;
37  
38          // code differs from Create because the descriptor check is deferred until after Prerelease.create
39          workingCopy = checkedWorkingCopy();
40          getLog().info("checking project ...");
41          revision = workingCopy.revision();
42          descriptor = Descriptor.create(version(), project, revision);
43          workingCopy.checkCompatibility(descriptor);
44          setTarget(archive.target(revision));
45          if (target.exists()) {
46              getLog().info("prerelease already exists: " + descriptor.getName());
47          } else {
48              maven = maven();
49              prerelease = Prerelease.create(maven, propertyArgs(), getLog(), descriptor, target);
50              archive.wipe(keep);
51              try {
52                  descriptor.check(world, project, allowSnapshots, allowPrereleaseSnapshots);
53              } catch (RuntimeException e) {
54                  throw e;
55              } catch (Exception e) {
56                  prerelease.target.scheduleRemove(getLog(), "build ok, but prerelease is not promotable: " + e.getMessage());
57                  getLog().debug(e);
58              }
59              if (snapshots) {
60                  prerelease.deploySnapshot(maven, getLog(), propertyArgs(), project);
61              }
62          }
63      }
64  }
