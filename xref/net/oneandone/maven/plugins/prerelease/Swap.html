18  import net.oneandone.maven.plugins.prerelease.core.Archive;
19  import net.oneandone.sushi.fs.Node;
20  import net.oneandone.sushi.fs.file.FileNode;
21  import org.apache.maven.plugins.annotations.Mojo;

23  import java.util.ArrayList;
24  import java.util.List;
25  import java.util.Set;

27  /**
28   * Wipes archives and moves prereleases to the next storage. You usually have two storages, primary and secondary.
29   * Useful when you use ram disks: use the ramdisk as primary storage, and a harddisk as secondary storage.
30   */
31  @Mojo(name = "swap", requiresProject = false)
32  public class Swap extends Base {
33      @Override
34      public void doExecute() throws Exception {
35          Set done;
36          List<FileNode> storages;
37          List<Node> archives;
38          Archive archive;
39          String relative;
40          FileNode dest;
41          FileNode storage;
42          int count;
43  
44          count = 0;
45          storages = storages();
46          // don't start at "-2" because we also want to wipe ...
47          for (int level = storages.size() - 1; level >= 0; level--) {
48              storage = storages.get(level);
49              getLog().info("checking storage: " + storage.getAbsolute());
50              archives = storage.find("*/*");
51              for (Node candidate : archives) {
52                  if (!candidate.isDirectory()) {
53                      continue;
54                  }
55                  relative = candidate.getRelative(storage);
56                  archive = Archive.tryOpen(directories(storages, relative));
57                  if (archive == null) {
58                      getLog().info("skipped because it is locked: " + relative);
59                  } else {
60                      try {
61                          archive.wipe(keep);
62                          for (FileNode src : archive.list().values()) {
63                              if (!src.join("prerelease.properties").readString().contains("prerelease=")) {
64                                  // This property was introduced in 1.6, together with multiple storage support
65                                  getLog().info("skipped -- prerelease version too old: " + relative);
66                                  continue;
67                              }
68                              if (level == storages.size() - 1) {
69                                  getLog().info("already in final storage: " + src);
70                              } else {
71                                  dest = storages.get(level + 1).join(relative, src.getName());
72                                  dest.getParent().mkdirsOpt();
73                                  src.move(dest);
74                                  getLog().info("swapped " + src.getAbsolute() + " -> " + dest.getAbsolute());
75                                  count++;
76                              }
77                          }
78                      } finally {
79                          archive.close();
80                      }
81                  }
82  
83              }
84          }
85          getLog().info(count + " archives swapped.");
86      }
87  
88      private static List<FileNode> directories(List<FileNode> storages, String relative) {
89          List<FileNode> result;
90  
91          result = new ArrayList<>();
92          for (FileNode storage : storages) {
93              result.add(storage.join(relative));
94          }
95          return result;
96      }
97  }
