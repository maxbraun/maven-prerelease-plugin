34  /** Transforms via DOM - transforming the model would loose all formatting. Maven's Release Plugin follows the same approach. */
35  public final class Transform {
36      /** @param tag null to skip scm transformation */
37      public static void adjustPom(FileNode pom, String expectedVersion, String nextVersion, String svnurl, String tag)
38              throws IOException, MojoExecutionException, SAXException, XmlException {
39          Document document;
40          Element element;
41          Selector selector;
42  
43          document = pom.readXml();
44          selector = new Selector();
45          selector.setNamespaceContext(new NamespaceContext() {
46              @Override
47              public String getNamespaceURI(String prefix) {
48                  if (!prefix.equals("M")) {
49                      throw new IllegalStateException("prefix: " + prefix);
50                  }
51                  return "http://maven.apache.org/POM/4.0.0";
52              }
53  
54              @Override
55              public String getPrefix(String namespaceURI) {
56                  throw new IllegalStateException();
57              }
58  
59              @Override
60              public Iterator getPrefixes(String namespaceURI) {
61                  throw new IllegalStateException();
62              }
63          });
64          element = selector.element(document, "/M:project/M:version");
65          if (!expectedVersion.equals(element.getTextContent().trim())) {
66              throw new MojoExecutionException("expected version " + expectedVersion + ", got " + element.getTextContent().trim());
67          }
68          element.setTextContent(nextVersion);
69          if (tag != null) {
70              scm(selector.elementOpt(document, "/M:project/M:scm/M:connection"), svnurl, tag);
71              scm(selector.elementOpt(document, "/M:project/M:scm/M:developerConnection"), svnurl, tag);
72              scm(selector.elementOpt(document, "/M:project/M:scm/M:url"), svnurl, tag);
73          }
74          writeRaw(document, pom);
75      }
76  
77      public static void writeRaw(Document document, FileNode file) throws IOException {
78          DOMImplementationLS ls;
79          LSSerializer serializer;
80          LSOutput output;
81          OutputStream stream;
82  
83          ls = (DOMImplementationLS) document.getImplementation().getFeature("LS", "3.0");
84          serializer = ls.createLSSerializer();
85          output = ls.createLSOutput();
86          stream = file.createOutputStream();
87          output.setByteStream(stream);
88          serializer.write(document, output);
89          stream.write('\n');
90          stream.close();
91      }
92  
93      private static void scm(Element element, String svnurl, String tag) throws MojoExecutionException {
94          String old;
95  
96          if (element == null) {
97              return;
98          }
99          old = element.getTextContent();
100         element.setTextContent(adjustScm(old, svnurl, tag));
101     }
102 
103     public static String adjustScm(String origStr, String svnurl, String tag) throws MojoExecutionException {
104         String str;
105 
106         if ("scm:svn:${project.scm.url}".equals(origStr)) {
107             // do not touch - implied by scm url.
108             return origStr;
109         }
110         str = origStr;
111         if (str.endsWith("/")) {
112             str = str.substring(0, str.length() - 1);
113         }
114         str = str.trim();
115         if (!str.endsWith(svnurl)) {
116             throw new MojoExecutionException("unexpected scm entry: " + origStr);
117         }
118         return str.substring(0, str.length() - svnurl.length()) + tag;
119     }

122     }
