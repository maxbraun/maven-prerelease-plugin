82      /**
83       * Creates an DefaultMaven instance, initializes it form parentRequest (in Maven, this is done by MavenCli - also by
84       * loading settings).
85       */
86      public void build(FileNode basedir, Map<String, String> userProperties, ExecutionListener theExecutionListener, boolean deployOnly,
87                        String ... goals) throws BuildException {
88          DefaultPlexusContainer container;
89          org.apache.maven.Maven maven;
90          MavenExecutionRequest request;
91          MavenExecutionResult result;
92          BuildException exception;
93          FilteringMojoExecutor mojoExecutor;
94  
95          request = DefaultMavenExecutionRequest.copy(parentSession.getRequest());
96          container = (DefaultPlexusContainer) parentSession.getContainer();
97          try {
98              maven = container.lookup(org.apache.maven.Maven.class);
99          } catch (ComponentLookupException e) {
100             throw new IllegalStateException(e);
101         }
102         request.setPom(basedir.join("pom.xml").toPath().toFile());
103         request.setProjectPresent(true);
104         request.setGoals(Arrays.asList(goals));
105         request.setBaseDirectory(basedir.toPath().toFile());
106         request.setUserProperties(merged(request.getUserProperties(), userProperties));
107         request.setExecutionListener(theExecutionListener);
108 
109         mojoExecutor = FilteringMojoExecutor.install(container, deployOnly ? "deploy" : null);
110         log.info("[" + basedir + "] mvn " + props(request.getUserProperties()) + Separator.SPACE.join(goals));
111         nestedOutputOn();
112         try {
113             result = maven.execute(request);
114         } finally {
115             nestedOutputOff();
116             mojoExecutor.uninstall();
117         }
118         exception = null;
119         for (Throwable e : result.getExceptions()) {
120             if (exception == null) {
121                 exception = new BuildException("build failed: " + e, e);
122             } else {
123                 exception.addSuppressed(e);
124             }
125         }
126         if (exception != null) {
127             throw exception;
128         }
129     }

133     /** @return with tailing space */
134     private static String props(Properties props) {
135         StringBuilder builder;
136 
137         builder = new StringBuilder();
138         for (Map.Entry<Object, Object> entry : props.entrySet()) {
139             builder.append("-D").append(entry.getKey());
140             if (!((String) entry.getValue()).isEmpty()) {
141                 builder.append('=').append(entry.getValue());
142             }
143             builder.append(' ');
144         }
145         return builder.toString();
146     }

158     /** Executes the deploy phase only, with the prerelease pom */
159     public void deployPrerelease(Log log, Map<String, String> propertyArgs, Prerelease prerelease) throws Exception {
160         PromoteExecutionListener listener;
161 
162         listener = new PromoteExecutionListener(prerelease, projectHelper, executionListener);
163         try {
164             build(prerelease.checkout, prerelease.descriptor.releaseProps(propertyArgs), listener, true, "deploy");
165         } catch (BuildException e) {
166             if (listener.isFirstSuccess()) {
167                 log.warn("Promote succeeded: your artifacts have been deployed, and your svn tag was created. ");
168                 log.warn("However, some optional deploy goals failed with this exception:");
169                 log.warn(e);
170                 log.warn("Thus, you can use your release, but someone should have a look at this exception.");
171             } else {
172                 throw e;
173             }
174         }
175     }

212     public MavenProject loadPom(FileNode file) throws ProjectBuildingException {
213         ProjectBuildingRequest request;
214         ProjectBuildingResult result;
215 
216         // session initializes the repository session in the build request
217         request = new DefaultProjectBuildingRequest(parentSession.getProjectBuildingRequest());
218         request.setRemoteRepositories(remoteRepositories);
219         request.setProcessPlugins(false);
220         request.setValidationLevel(ModelBuildingRequest.VALIDATION_LEVEL_MINIMAL);
221         request.setSystemProperties(System.getProperties());
222         //If you don't turn this into RepositoryMerging.REQUEST_DOMINANT the dependencies will be resolved against Maven Central
223         //and not against the configured repositories. The default of the DefaultProjectBuildingRequest is
224         // RepositoryMerging.POM_DOMINANT.
225         request.setRepositoryMerging(ProjectBuildingRequest.RepositoryMerging.REQUEST_DOMINANT);
226         request.setResolveDependencies(false);
227         result = builder.build(file.toPath().toFile(), request);
228         return result.getProject();
229     }

233     // Marks nested output. I'd love to just indent it by 4 spaces, but there's no single place to do this.
234     // * it's too late to configure a LoggerFactory because Maven has already setup most of the components,
235     //   and they'll be re-used by the build command
236     // * every component has it's own Logger instance, that holds the target output stream; there's no
237     //   way to get hold of all these loggers
238     // * System.setOut does not affect the output streams already stored in the component loggers
