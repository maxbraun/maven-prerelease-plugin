43  /**
44   * The archive stores prereleases for one given groupId/artifactId.
45   *  Prerelease directories have the svn revision number as their directory name; it may be a symlink to an arbitrary location
46   */
47  //
48  // ~/.m2/prereleases          <- default storage for all archives. This is the primary storage, swap also uses a secondary storage.
49  //   groupId/artifactId.LOCK  <- optional, indicates that a process operates on this archive
50  //   groupId/artifactId/      <- archive directory
51  //    |- revision1           <- prerelease directory, ready to promote; promoting the prerelease removes this directory
52  //    |     |- tags
53  //    |     |    - <tagname>
54  //    |     |- artifacts
55  //    |      - prerelease.properties
56  //    |- revision2
57  //    :
58  //    |- REMOVE              <- a renamed prerelease directory
59  //    :     :                   optional - only when the last create call failed (because the mvn call failed) or promote succeeded
60  //    :     |- CAUSE         <- why this directory is to be removed
61  //    :     :
62  
63  public class Archive implements AutoCloseable {
64      public static List<FileNode> directories(List<FileNode> storages, MavenProject project) {
65          List<FileNode> directories;
66  
67          directories = new ArrayList<>(storages.size());
68          for (FileNode storage : storages) {
69              directories.add(storage.join(project.getGroupId(), project.getArtifactId()));
70          }
71          return directories;
72      }

83          Archive archive;

101     public Target target(long revision) {
102         String name;
103         FileNode prerelease;
104 
105         name = Long.toString(revision);
106         for (int i = directories.size() - 1; i >= 0; i--) {
107             prerelease = directories.get(i).join(name);
108             if (i == 0 || prerelease.exists()) {
109                 return new Target(prerelease, revision);
110             }
111         }
112         throw new IllegalStateException();
113     }

115     /** */
116     public TreeMap<Long, FileNode> list() throws ListException, DirectoryNotFoundException {
117         TreeMap<Long, FileNode> result;
118         long revision;
119 
120         result = new TreeMap<>();
121         for (FileNode directory : directories) {
122             if (directory.exists()) {
123                 for (FileNode prerelease : directory.list()) {
124                     if (!prerelease.getName().equals(Target.REMOVE)) {
125                         revision = Long.parseLong(prerelease.getName());
126                         result.put(revision, prerelease);
127                     }
128                 }
129             }
130         }
131         return result;
132     }
133 
134     public long latest() throws ListException, DirectoryNotFoundException {
135         return list().lastKey();
136     }

140     private FileNode lockFile() {
141         FileNode primary;
142 
143         primary = directories.get(0);
144         return primary.getParent().join(primary.getName() + ".LOCK");
145     }
146 
147     /**
148      * @param timeout in seconds; -1 to try only once and never wait.
149      * @param log may be null
150      */
151     private void open(int timeout, Log log) throws IOException {
152         FileNode file;
153         int seconds;
154 
155         if (opened) {
156             throw new IllegalStateException();
157         }
158         file = lockFile();
159         try {
160             seconds = 0;
161             while (true) {
162                 // every time - if someone wiped the primary storage directory
163                 file.getParent().mkdirsOpt();
164                 try {
165                     file.mkfile();
166                     OnShutdown.get().deleteAtExit(file);
167                     opened = true;
168                     file.writeString(Integer.toString(pid()));
169                     if (log != null) {
170                         log.debug("locked for pid " + pid());
171                     }
172                     return;
173                 } catch (MkfileException e) {
174                     if (seconds > timeout) {
175                         if (log != null) {
176                             log.warn("Lock timed out after " + seconds + "s.");
177                         }
178                         throw e;
179                     }
180                     if (seconds % 10 == 0) {
181                         if (log != null) {
182                             log.info("Waiting for " + file + ": " + seconds + "s");
183                             log.debug(e);
184                         }
185                     }
186                     seconds++;
187                     Thread.sleep(1000);
188                 }
189             }
190         } catch (InterruptedException e) {
191             if (log != null) {
192                 log.warn("interrupted");
193             }
194         }
195     }
196 
197     @Override
198     public void close() throws Exception {
199         FileNode file;
200 
201         if (!opened) {
202             throw new IllegalStateException("not opened");
203         }
204         if (closed) {
205             throw new IllegalStateException("already closed");
206         }
207         file = lockFile();
208         file.deleteFile();
209         // because another thread waiting for this lock might create this file again.
210         // The shutdown hook must not delete the file created by this other thread.
211         OnShutdown.get().dontDeleteAtExit(file);
212         closed = true;
213     }

217     private static int pid = 0;
218 
219     public static int pid() {
220         String str;
221         int idx;
222 
223         if (pid == 0) {
224             // see http://blog.igorminar.com/2007/03/how-java-application-can-discover-its.html?m=1
225             str = ManagementFactory.getRuntimeMXBean().getName();
226             idx = str.indexOf('@');
227             if (idx == -1) {
228                 throw new IllegalStateException("cannot guess pid from " + str);
229             }
230             pid = Integer.parseInt(str.substring(0, idx));
231         }
232         return pid;
233     }

235     /** @return true when a changes file was found */
236     public static boolean adjustChangesOpt(FileNode workingCopy, String version) throws XmlException, IOException, SAXException {
237         ChangesXml changes;
238 
239         try {
240             changes = ChangesXml.load(workingCopy);
241         } catch (FileNotFoundException e) {
242             return false;
243         }
244         changes.releaseDate(version, new Date());
245         changes.save();
246         return true;
247     }

249     /**
250      * @param keep number of prereleases after this method
251      */
252     public void wipe(int keep) throws IOException {
253         TreeMap<Long, FileNode> prereleases;
254         FileNode d;
255 
256         if (keep < 1) {
257             throw new IllegalArgumentException("keep " + keep);
258         }
259         for (FileNode directory : directories) {
260             d = directory.join(Target.REMOVE);
261             if (d.isDirectory()) {
262                 d.deleteTree();
263             }
264         }
265         prereleases = list();
266         while (prereleases.size() > keep) {
267             prereleases.remove(prereleases.firstKey()).deleteTree();
268         }
269     }
