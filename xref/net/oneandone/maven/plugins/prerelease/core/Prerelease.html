53      public static Prerelease create(Maven maven, Map<String, String> propertyArgs, Log log, Descriptor descriptor, Target target) throws Exception {
54          Prerelease prerelease;
55          FileNode tags;
56          FileNode checkout;
57          String tagname;
58          String tagbase;
59          int idx;
60  
61          log.info("creating un-committed tag ...");
62          if (descriptor.svnTag.endsWith("/")) {
63              throw new IllegalArgumentException(descriptor.svnTag);
64          }
65          idx = descriptor.svnTag.lastIndexOf('/');
66          if (idx == -1) {
67              throw new IllegalArgumentException(descriptor.svnTag);
68          }
69          tagbase = descriptor.svnTag.substring(0, idx);
70          tagname = descriptor.svnTag.substring(idx + 1);
71          target.create();
72          try {
73              tags = target.join("tags");
74              checkout = tags.join(tagname);
75              log.debug(target.svnLauncher("checkout", "--depth=empty", tagbase, tags.getAbsolute()).exec());
76              log.debug(target.svnLauncher("copy", "-r" + descriptor.revision, descriptor.svnOrig, checkout.getAbsolute()).exec());
77              prerelease = new Prerelease(target, checkout, descriptor);
78              prerelease.descriptor.save(target);
79              Transform.adjustPom(prerelease.checkout.join("pom.xml"), descriptor.previous, descriptor.project.version,
80                      descriptor.svnOrig, descriptor.svnTag);
81              Archive.adjustChangesOpt(prerelease.checkout, prerelease.descriptor.project.version);
82              prerelease.create(maven, propertyArgs);
83              log.info("created prerelease in " + prerelease.target);
84          } catch (Exception e) {
85              target.scheduleRemove(log, "create failed: " + e.getMessage());
86              throw e;
87          }
88          return prerelease;
89      }

146     public FileNode prepareOrigCommit(Log log) throws IOException, XmlException, SAXException, MojoExecutionException {
147         FileNode result;
148         ChangesXml changes;
149 
150         result = checkout.getWorld().getTemp().createTempDirectory();
151         Subversion.sparseCheckout(log, result, descriptor.svnOrig, "HEAD", true);
152         try {
153             changes = ChangesXml.load(result);
154         } catch (FileNotFoundException e) {
155             log.info("no changes.xml to adjust.");
156             changes = null;
157         }
158 
159         Subversion.launcher(result, "lock", "pom.xml");
160         if (changes != null) {
161             Subversion.launcher(result, "lock", ChangesXml.PATH);
162         }
163 
164         // make sure the version we've locked is what we will modify:
165         // (or in other words: make sure we see possible changes that were committed between checkout and lock)
166         Subversion.launcher(result, "up");
167 
168         Transform.adjustPom(result.join("pom.xml"), descriptor.previous, descriptor.next, null, null);
169         if (changes != null) {
170             changes.releaseDate(descriptor.project.version, new Date());
171             changes.save();
172         }
173         // svn up does not fail for none-existing files!
174         return result;
175     }

200     public void promote(Log log, Map<String, String> propertyArgs, String createTagMessage, String revertTagMessage, String nextIterationMessage, Maven maven) throws Exception {
201         FileNode origCommit;
202 
203         log.info("promoting revision " + descriptor.revision + " to " + descriptor.project);
204         origCommit = prepareOrigCommit(log);
205         try {
206             promoteLocked(log, propertyArgs, createTagMessage, revertTagMessage, nextIterationMessage, origCommit, maven);
207         } catch (Throwable e) { // CAUTION: catching exceptions is not enough -- in particular, out-of-memory during upload is an error!
208             try {
209                 origUnlock(origCommit);
210             } catch (Exception nested) {
211                 e.addSuppressed(nested);
212             }
213             throw e;
214         }
215         origUnlock(origCommit);
216         log.info("SUCCESS: released " + descriptor.project);
217     }

219     private void origUnlock(FileNode origCommit) {
220         Subversion.launcher(origCommit, "unlock" , "pom.xml");
221         if (origCommit.join(ChangesXml.PATH).exists()) {
222             Subversion.launcher(origCommit, "unlock" , ChangesXml.PATH);
223         }
224     }

226     /** commit before deploy - because if deployment fails, we can reliably revert the commit. */
227     private void promoteLocked(Log log, Map<String, String> propertyArgs, String commitTagMessage, String revertTagMessage, String commitNextMessage,
228             FileNode origCommit, Maven maven) throws Exception {
229         FileNode installed;
230 
231         commit(log, renderMessage(commitTagMessage));
232         try {
233             maven.deployPrerelease(log, propertyArgs, this);
234         } catch (Exception e) {
235             log.info("deployment failed - reverting tag");
236             revertCommit(log, renderMessage(revertTagMessage));
237             target.scheduleRemove(log, "deployment failed (tag has been reverted): " + e.getMessage());
238             throw e;
239         }
240 
241         // local install
242         installed = descriptor.project.localRepo(maven);
243         installed.deleteTreeOpt();
244         artifacts().move(installed);
245 
246         try {
247             log.info("Update pom and changes ...");
248             log.debug(Subversion.launcher(origCommit, "commit", "-m", renderMessage(commitNextMessage)).exec());
249             origCommit.deleteTree();
250             // Move prerelease directory into REMOVE directory because it's invalid now:
251             // tag was committed, and artifacts have been deployed. It's not removed immediately to make
252             // distribution file available locally.
253             target.scheduleRemove(log, "prerelease has been promoted");
254         } catch (Exception e) {
255             log.warn("Promote succeeded: your artifacts have been deployed, and your svn tag was created. ");
256             log.warn("However, some post-release step failed with this exception:");
257             log.warn(e);
258             log.warn("Thus, you can use your release, but someone should have a look at this exception.");
259         }
260     }

262     public String renderMessage(String message) throws SubstitutionException {
263         Map<String, String> variables;
264 
265         variables = new HashMap<>();
266         variables.put("revision", Long.toString(descriptor.revision));
267         variables.put("release", descriptor.project.toString());
268         return Substitution.ant().apply(message, variables);

273     public void artifactFiles(MavenProject project, MavenProjectHelper projectHelper) throws IOException {
274         FileNode file;
275         String type;
276         String classifier;
277         String[] tmp;
278 
279         for (Map.Entry<FileNode, String[]> entry : artifactFiles().entrySet()) {
280             file = entry.getKey();
281             tmp = entry.getValue();
282             classifier = tmp[0];
283             type = tmp[1];
284             if ("pom".equals(type) && !project.getPackaging().equals("pom")) {
285                 // ignored
286             } else {
287                 if (classifier == null) {
288                     project.getArtifact().setFile(file.toPath().toFile());
289                 } else {
290                     projectHelper.attachArtifact(project, type, classifier, file.toPath().toFile());
291                 }
292             }
293         }
294     }

296     /** @return Mapping of Artifact File to [ classified, type ]. Classifier is null for the main artifact */
297     public Map<FileNode, String[]> artifactFiles() throws IOException {
298         FileNode artifacts;
299         String name;
300         String str;
301         String type;
302         String classifier;
303         Map<FileNode, String[]> result;
304 
305         artifacts = artifacts();
306         result = new HashMap<>();
307         for (FileNode file : artifacts.list()) {
308             name = file.getName();
309             if (name.endsWith(".md5") || name.endsWith(".sha1") || name.endsWith(".asc")
310                     || /* maven 3.0 */ name.equals("_maven.repositories") || /* maven 3.1 */ name.equals("_remote.repositories")) {
311                 // skip
312             } else {
313                 type = file.getExtension();
314                 str = name.substring(0, name.length() - type.length() - 1);
315                 str = Strings.removeLeft(str, descriptor.project.artifactId + "-");
316                 str = Strings.removeLeft(str, descriptor.project.version);
317                 if (str.isEmpty()) {
318                     classifier = null;
319                 } else {
320                     classifier = Strings.removeLeft(str, "-");
321                 }
322                 result.put(file, new String[] { classifier, type });
323             }
324         }
325         return result;
326     }
