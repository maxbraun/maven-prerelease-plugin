37      protected String archiveRoot;

40       * Timeout in seconds for locking a prerelease archive.

43      protected int lockTimeout;

45      @Component
46      private ProjectBuilder projectBuilder;
47  
48      @Component
49      private MavenProjectHelper projectHelper;
50  
51      /**

54      @Parameter(property = "repositorySystemSession", readonly = true)
55      private RepositorySystemSession repositorySession;

57      @Parameter(property = "project.remoteArtifactRepositories", readonly = true)
58      private List<ArtifactRepository> remoteRepositories;
59

61      protected MavenSession session;

67      }

72          try {

75              throw e;
76          } catch (Exception e) {
77              throw new MojoExecutionException("plugin failed: " + e.getMessage(), e);
78          }
79      }
80  
81      public abstract void doExecute() throws Exception;
82

84          return new Maven(world, session, session.getRequest().getExecutionListener(), projectHelper, repositorySession, projectBuilder, remoteRepositories);
85      }
