29  /**
30   * Promotes a prerelease by commuting the tag and deploying its artifact(s).
31   *
32   * Execute this goal in the svn working directory of the project you want to release and make sure you have created a prerelease.
33   *
34   * This goal commits the tag of the current prerelease and deploys the respective artifact(s). It also updates (in your svn trunk or branch
35   * on the svn server) the project version in pom.xml to the next development version, and it updates your changes file (if you have one).
36   * In your svn working directory, it runs "svn up" to get the adjusted pom and changes files.
37   *
38   * Artifacts are deployed by invoking all plugin goals tied to the deploy phase. The first goal is mandatory, because it's configured by
39   * the packaging. If it fails, promotion fails. All other goals (e.g. email notifications) are optional: they may fail (yielding the
40   * respective warning), but they don't cause the promote goal to fail.
41   *
42   * Error handling. In contrast to Maven's Release Plugin there's no rollback goal: when promote fails, artifacts and tags will be properly
43   * removed. (If Tag creation fails, this goal fails with an error. If artifact deployment fails, the svn tag will be deleted from
44   * the repository and the goal fails with an error. If optional promote goal fail, the plugins succeeds, but it issues a warning.)
45   *
46   * Note that the deployment date of artifacts in the repository may be later that the release date in the changes.xml.
47   */
48  @Mojo(name = "promote")
49  public class Promote extends ProjectBase {
50      /**
51       * Message for svn commit of the new tag.
52       */
53      @Parameter(property = "prerelease.createTagMessage", defaultValue =
54              "Prerelease ${revision} promoted to release ${release}.")
55      protected String createTagMessage;
56  
57      /**
58       * Svn commit message when reverting the tag.
59       */
60      @Parameter(property = "prerelease.revertTagMessage", defaultValue =
61              "Reverting tag for release ${release} because the deployment failed.")
62      protected String revertTagMessage;
63  
64      /**
65       * Message for svn commit to start new development iteration.
66       */
67      @Parameter(property = "prerelease.nextIterationMessage", defaultValue =
68              "Prerelease ${revision} promoted to release ${release}, starting next development iteration.")
69      protected String nextIterationMessage;
70  
71      public void doExecute(Archive archive) throws Exception {
72          WorkingCopy workingCopy;
73          long revision;
74          Prerelease prerelease;
75  
76          workingCopy = checkedWorkingCopy();
77          revision = workingCopy.revision();
78          setTarget(archive.target(revision));
79          prerelease = target.loadOpt();
80          if (prerelease == null) {
81              throw new MojoExecutionException("no prerelease for revision " + revision);
82          }
83          prerelease.promote(getLog(), createTagMessage, revertTagMessage, nextIterationMessage, maven());
84          workingCopy.update(getLog());
85      }
86  }
