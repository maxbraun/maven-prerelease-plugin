25  /**
26   * Creates a prerelease with an uncommitted tag and undeployed artifacts. Creating a prerelease is the first step to create a release;
27   * once you have a prerelease, you can quicky get a release by promoting the prerelease.
28   *
29   * Execute this in the svn working directory of your project. Basically, this goal runs checks, creates an uncommitted tag of your working
30   * directory and invokes "mvn clean deploy" in it. Maven is invoked with an alternative deployment repository pointing into a subdirectory
31   * of this prerelease. In addition, the "performRelease" property is defined to get the same profiles activation you get during a
32   * "release:promote" call,
33   *
34   * When successful, prereleases are stored in the configured archive directory, previous prerelease are wiped.
35   *
36   * Checks executed by this goal:
37   *
38   * 1) no uncommitted changes in your working directory.
39   *
40   * 2) no pending updates up to the last modified revision
41   *
42   * 3) no snapshot dependencies or snapshot parent pom.
43   */
44  @Mojo(name = "create")
45  public class Create extends ProjectBase {
46      @Override
47      public void doExecute(Archive archive) throws Exception {
48          WorkingCopy workingCopy;
49          Descriptor descriptor;
50  
51          workingCopy = checkedWorkingCopy();
52          setTarget(archive.target(workingCopy.revision()));
53          if (target.exists()) {
54              throw new MojoExecutionException("prerelease already exists: " + workingCopy.revision());
55          }
56          descriptor = checkedDescriptor(workingCopy);
57          Prerelease.create(maven(), getLog(), descriptor, target);
58      }
59  }
