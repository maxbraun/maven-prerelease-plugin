34  /**
35   * Find an delete stale locks. JVM crashes or kill -9 does not properly cleanup locks. This goals repairs them by looking for locks
36   * from processes that no longer exists.
37   *
38   * Note that this locking cleanup should run in a single, separate process to avoid running into its own locking problems. It
39   * must not be included in the lock file creating itself because there are multiple processes involved.
40   */
41  @Mojo(name = "locksmith", requiresProject = false)
42  public class Locksmith extends Base {
43      /**
44       * Fail when stale locks are detected?
45       */
46      @Parameter(property = "prerelease.locksmith.fail")
47      private boolean fail = false;
48  
49      /**
50       * Delete stale locks?
51       */
52      @Parameter(property = "prerelease.locksmith.delete")
53      private boolean delete = false;
54  
55      private static final long COUNT = 3;
56      private static final long HOUR = 1000L * 60 * 60;
57  
58      @Override
59      public void doExecute() throws Exception {
60          Map<String, Long> started;
61          FileNode root;
62          String pid;
63          Long time;
64          int errors;
65          List<Node> locks;
66  
67          started = startedMap();
68          errors = 0;
69          root = world.file(archiveRoot);
70          if (!root.exists()) {
71              return;
72          }
73          locks = root.find("*/*.LOCK");
74          for (Node file : locks) {
75              pid = file.readString();
76              if (pid.trim().isEmpty()) {
77                  getLog().info(file + ": old lock file format");
78                  if (System.currentTimeMillis() - file.getLastModified() < HOUR * COUNT) {
79                      getLog().info("> younger than " + COUNT + " hours - considered ok");
80                      continue;
81                  }
82              } else {
83                  time = started.get(pid);
84                  if (time == null) {
85                      getLog().info(file + ": stale lock - no process with id " + pid);
86                  } else if (file.getLastModified() < time) {
87                      getLog().info(file + ": stale lock - process with id " + pid + " younger than lock file");
88                  } else {
89                      getLog().debug(file + " ok");
90                      continue;
91                  }
92              }
93              errors++;
94              if (delete) {
95                  getLog().info("deleting " + file);
96                  file.deleteFile();
97              }
98          }
99          getLog().info("locks checked: " + locks.size() + ", locks stale: " + errors);
100         if (errors > 0 && fail) {
101             throw new MojoExecutionException("stale locks: " + errors);
102         }
103     }
104 
105     private static final SimpleDateFormat TODAY = new SimpleDateFormat("HH:mm:ss");
106     private static final SimpleDateFormat OTHER = new SimpleDateFormat("MMM dd", Locale.US);
107 
108     private Map<String, Long> startedMap() throws IOException {
109         Map<String, Long> result;
110         boolean first;
111         int idx;
112         String pid;
113         String startedStr;
114         Date startedDate;
115 
116         result = new HashMap<>();
117         first = true;
118         for (String line : Separator.RAW_LINE.split(((FileNode) world.getWorking()).exec("ps", "ax", "--format", "pid,start"))) {
119             if (first) {
120                 if (!line.contains("PID")) {
121                     throw new IllegalStateException(line);
122                 }
123                 first = false;
124             } else {
125                 line = line.trim();
126                 idx = line.indexOf(' ');
127                 if (idx == -1) {
128                     throw new IllegalStateException(line);
129                 }
130                 pid = line.substring(0, idx);
131                 startedStr = line.substring(idx + 1).trim();
132                 try {
133                     startedDate = (startedStr.indexOf(':') == -1 ? OTHER: TODAY).parse(startedStr);
134                 } catch (ParseException e) {
135                     throw new IllegalStateException("invalid date in line " + line, e);
136                 }
137                 result.put(pid, startedDate.getTime());
138             }
139         }
140         return result;
141     }
